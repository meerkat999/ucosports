CREATE TABLE `ucosportstournament`.`player` (
  `idplayer` INT NOT NULL AUTO_INCREMENT,
  `firstnameplayer` VARCHAR(45) NULL,
  PRIMARY KEY (`idplayer`));

CREATE TABLE `ucosportstournament`.`sport` (
  
`sport_id` INT NOT NULL AUTO_INCREMENT,
  
`sport_name` VARCHAR(45) NOT NULL, `sport_number_players` 
INT NOT NULL, `sport_number_times` INT NOT NULL, `sport_time_times` 
INT NOT NULL,
  PRIMARY KEY (`sport_id`));
  
CREATE TABLE `ucosportstournament`.`sportstatistic` (
  `sportstatistic_id` INT NOT NULL AUTO_INCREMENT,
  `sportstatistic_name` VARCHAR(45) NOT NULL,
  `sport_id` INT NOT NULL,
  PRIMARY KEY (`sportstatistic_id`),
  INDEX `fk_sport_id_idx` (`sport_id` ASC),
  CONSTRAINT `fk_sport_id`
    FOREIGN KEY (`sport_id`)
    REFERENCES `ucosportstournament`.`sport` (`sport_id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
